### 检测窗口是否可以在屏幕完全显示

```python

import sys  
from PySide6.QtWidgets import QApplication, QWidget, QVBoxLayout, QHBoxLayout
from PySide6.QtGui import QColor
from PySide6.QtCore import Qt, QRect


class Demo(QWidget):
    
    def checkVisible(self):
        """ 通过 Rect """
        rect = self.frameGeometry()
        
        screen = QApplication.screenAt(rect.center())
        if not screen:
            screen = QApplication.primaryScreen()
        available = screen.availableGeometry()
        
        """ 通过 Point """
        rect = QRect(self.pos(), self.size())
        
        return available.contains(rect)

    def overflowValue(self, rect: QRect, available: QRect):
        return [
            max(0, available.left() - rect.left()),
            max(0, available.top() - rect.top()),
            max(0, available.right() - rect.right()),
            max(0, available.bottom() - rect.bottom())
        ]
```
